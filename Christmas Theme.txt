        // Add dynamic snowflakes (replaces "stars")
        function createStars() {
            const container = document.body;
            for (let i = 0; i < 18; i++) { // reduced from 40 to 18
                const star = document.createElement('div');
                star.className = 'snowflake';
                star.textContent = '❆';
                star.style.left = Math.random() * 100 + '%';
                star.style.top = Math.random() * 100 + '%';
                const size = 8 + Math.random() * 18;
                star.style.fontSize = size + 'px';
                star.style.opacity = 0.6 + Math.random() * 0.4;
                star.style.animationDelay = (Math.random() * 3) + 's';
                star.style.color = ['#ffffff', '#B30000', '#0B6623'][Math.floor(Math.random() * 3)];
                container.appendChild(star);
            }
        }

        // Create snow particle puffs with red & green accent puffs
        function createParticles() {
            const container = document.querySelector('.particles');
            for (let i = 0; i < 24; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = -Math.random() * 40 + 'vh';
                particle.style.animationDelay = Math.random() * 15 + 's';
                const colorOptions = ['#ffffff', '#B30000', '#0B6623'];
                particle.style.backgroundColor = colorOptions[Math.floor(Math.random() * colorOptions.length)];
                particle.style.opacity = 0.8 + Math.random() * 0.2;
                particle.style.width = (4 + Math.round(Math.random() * 6)) + 'px';
                particle.style.height = particle.style.width;
                container.appendChild(particle);
            }
        }

        createStars();
        createParticles();

        // Popup show-once logic

            document.getElementById('popupCloseBtn').onclick = function() {
                document.getElementById('popupUpdate').style.display = 'none';
                document.getElementById('popupOverlay').style.display = 'none';
                localStorage.setItem('popupUpdateRead', 'true');
            };
        });
    </script>
